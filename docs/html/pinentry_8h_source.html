117  int parent_wid;

173  char *repeat_ok_string;
174

188  char *quality_bar_tt;

198  char *genpin_tt;

207  char *formatted_passphrase_hint;

293 /* The pinentry command handler type processes the pinentry request
294  PIN. If PIN->pin is zero, request a confirmation, otherwise a PIN
295  entry. On confirmation, the function should return TRUE if
296  confirmed, and FALSE otherwise. On PIN entry, the function should
297  return -1 if an error occurred or the user cancelled the operation
298  and 1 otherwise. */
